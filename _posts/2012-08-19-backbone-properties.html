--- layout: post title: backbone-properties categories: ru --- В Jquery используются удобные методы-свойства, например. 

jElement.val(); // работает как геттер
jElement.val('some value'); // работает как сеттер
В Backbone для доступа к различным свойствам используются методы get и set, в итоге во вьюшках приходиться писать 

this.model.set('myattr');
Плагин backbone-properties добавляет метод createProperty в модель, который создаёт удобные методы доступа к атрибутам модели. Пример объяснит лучше: 

var User = Backbone.Model.extend({
  initialize: function() {
    // можно создать одно свойсто
    this.createProperty('name');

    // или сразу массив
    this.createProperty([
      'age',
      'phone',
      'city'
    ]);
  }
});

var albert = new User({
  name: 'Albert',
  age: 20,
  phone: '555-55-55',
  city: 'Tokyo'
});


albert.name(); // return 'Albert'

albert.name('Ivan');
albert.name(); // return 'Ivan'

// можно вторым параметром в сеттере указывать стандартные для Model.set опции
albert.age(22, {silent: true}); // не создаст событие change
